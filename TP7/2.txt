FONCTION double minNormeAccel(entier_double min,entier_double norme)
	double a
	SI min < norme
		a = min
	SINON 
		a = norme
	FIN_SI

FIN_FONCTION

	entier_double x,y,z
	entier_double norme, min
	entier i
